# 系统服务配置文件
# 复制到: /etc/systemd/system/resume-system.service

[Unit]
Description=Resume Information Extraction System
After=network.target

[Service]
Type=simple
User=root
WorkingDirectory=/root/resume_system
Environment="PATH=/usr/local/bin:/usr/bin:/bin"
ExecStart=/usr/bin/python3 app.py
Restart=always
RestartSec=10

# 日志
StandardOutput=append:/var/log/resume_system.log
StandardError=append:/var/log/resume_system_error.log

[Install]
WantedBy=multi-user.target
